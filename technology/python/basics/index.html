    字典排序

    name = ['达拉斯', '大幅', '热个', '个胡', '圣啊']
age = [31, 32, 33, 34, 35]
index = [5, 2, 3, 4, 1]
diclist = []
for i in range(len(name)):
    dic = {'name': name[i], 'age': age[i], 'index': index[i]}
    diclist.append(dic)
# 字典排序
c = sorted(diclist, key=lambda diclist: diclist['index'], reverse=False)
print(c)
